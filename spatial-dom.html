تعديل الصورة في المجال المكاني

التنعيم

مرشح المتوسط الحسابي


img = imread('cameraman.tif');
img = im2double(img);
[rows,cols] = size(img);
%قناع المرشح
msk = 1/9*ones(3,3);
outImg = zeros(size(img));

for r = 2:rows-1
 for c = 2:cols-1
  win = img(r-1:r+1,c-1:c+1);
  outImg(r,c) = sum(msk(:).*win(:));
 end
end

إزالة الضجيج

ضجيج الملح والفلفل (مرشح القيمة الوسطى)


img = imread('noisyimg.png');
img = im2double(img);
[rows,cols] = size(img);
outImg = zeros(size(img));

for r = 2:rows-1
 for c = 2:cols-1
  win = img(r-1:r+1,c-1:c+1);
  winVals = sort(win(:));
  %وسيط عناصر النافذة المنزلقة
  midVal = winVals(5);
  outImg(r,c) = midVal;
 end
end

كشف الحواف

اعتماداً على المشتق الأول (مرشح بريويت)


img = imread('cameraman.tif');
img = im2double(img);
%مركبة بريويت الأفقية
preX = [-1 0 1;-1 0 1;-1 0 1];
%مركبة بريويت العمودية
preY = [-1 -1 -1;0 0 0;1 1 1];
[rows,cols] = size(img);

for r = 2:rows-1
 for c = 2:cols-1
  win = img(r-1:r+1,c-1:c+1);
  %كشف الحواف العمودية
  edgesY(r,c) = sum(win(:).*preX(:));
  %كشف الحواف الأفقية
  edgesX(r,c) = sum(win(:).*preY(:));
 end
end

%قوة الحواف
edges = sqrt(edgesX.^2+edgesY.^2);
%اتجاه الحواف
theta = atan(edgesY./edgesX );

اعتماداً على المشتق الثاني (لابلاسيان)


img = imread('cameraman.tif');
img = im2double(img);
%قناع مرشح لابلاسيان
lapMsk = [0 -1 0;-1 4 -1;0 -1 0];
[rows,cols] = size(img);
edges = zeros(size(img));
for r = 2:rows-1
 for c = 2:cols-1
  win = img(r-1:r+1,c-1:c+1);
  edges(r,c) = sum(win(:).*lapMsk(:));
 end
end

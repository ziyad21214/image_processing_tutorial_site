توليد وعرض هيستوغرام الصورة


img = imread('cameraman.tif');
hist = zeros(1,256);
for i = 1:255
 pixCount = length(find(img==i));
 hist(i+1) = pixCount;
end
bar(hist)

img = imread('cameraman.tif');
img = im2double(img);
minI = prctile(img(:),2);
maxI = prctile(img(:),98);
newImg = (img-minI)./(maxI-minI);

تسوية الهيستوغرام


img = imread('cameraman.tif');
%إيجاد الهيستوغرام الأصلي
hist = imhist(img);

%إيجاد الجمع التراكمي
cumSum = zeros(size(hist));
cumSum(1) = hist(1);
for i = 2:length(hist)
 cumSum(i) = hist(i)+cumSum(i-1);
end

%التنسيب وإيجاد السويات الجديدة
normSum = (cumSum-cumSum(1))./...
(cumSum(end)-cumSum(1))*255;
normSum = floor(normSum);

%الحصول على الصورة الجديدة
newImg = img;
for i = 1:length(normSum)
 newImg(img==i-1) = normSum(i);
end

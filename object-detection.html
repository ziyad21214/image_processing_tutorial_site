كشف الكائنات والقياسات

كشف الكائنات وإيجاد خصائصها (قياساتها)


img = imread('coins.png');
bw_img = im2bw(img);
%ردم الثقوب
bw_img = imfill(bw_img, 'holes');
%تسمية الكائنات وإيجاد عددها
[lbls, objs] = bwlabel(bw_img);

%إيجاد خصائص الكائنات
stats = regionprops(lbls,'all');
%المساحة
areas = cat(1, stats.Area)
%مراكز الثقل
cents = cat(1, stats.Centroid)
%المربع المحيط الأدنى
boundBox = cat(1, stats.BoundingBox)
%المحيط
per = cat(1, stats.Perimeter)
%التراص
comp = (per.^2)./(4*pi*areas)
%الاختلاف المركزي
ecc = cat(1, stats.Eccentricity)
%الامتداد
ext = cat(1, stats.Extent)
